#include <bits/stdc++.h>
using namespace std;
int main() {
    int n ; cin>>n;
    int arr[n] ;
     for( int i=0;i<n;i++){
        cin>>arr[i];
     }
     int k, m=0; cin>>k;
     map<int , int> mp;
    int i =0 , j =0 ;

    while(j<n){
        mp[arr[j]]++;
        if(mp.size()==j-i+1){
            m=max(m,j-i+1);
            j++;
        }
        else if(mp.size()<j-i+1){
            while(mp.size()<j-i+1){
                mp[arr[i]]--;
                if(mp[arr[i]]==0){
                    mp.erase(arr[i]);
                }
                i++;
            }
            j++;
        }
    }
    cout<<m;
}
